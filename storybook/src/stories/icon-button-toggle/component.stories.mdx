### Example Usage

```tsx
import { IconButtonToggle } from '@equinor/fusion-react-icon-button-toggle';

  // with Fusion icon
  <IconButtonToggle offIcon="wifi_off" onIcon="wifi" props />

  // with custom icons
  <IconButtonToggle props >
    <img slot="offIcon" src />
    <img slot="onIcon" src />
  </IconButtonToggle>

  // with custom images
  <IconButtonToggle props >
    <svg slot="offIcon"> </svg>
    <svg slot="onIcon"> </svg>
  </IconButtonToggle>
```
