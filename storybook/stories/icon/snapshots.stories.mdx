import { Meta, Story, Canvas, ArgsTable, Description } from '@storybook/addon-docs';
import readme from '@equinor/fusion-react-icon/README.md';
import { Icon, iconNames } from '@equinor/fusion-react-icon';

export const Template = ({ icon, size, color }) => {
  const style = {
    fontSize: size,
    color,
  };
  return <Icon style={style} icon={icon}/>;
};

export const Container = ({children}) => (
  <div style={{display: 'flex', gap: '10px', flexWrap: 'wrap'}}>
    {children}
  </div>
);

export const IconInfo = (props) => {
  const style = {
    display: 'inline-flex',
    flexFlow: 'column',
    alignItems: 'center',
    border: '1px solid',
    borderRadius: '.25rem',
    minWidth: '175px'
  };
  return (
    <div style={style}>
      <Template {...props} />
      <span>{props.icon}</span>
    </div>
  );
}

<Meta title="Snapshots/Icon"/>

### Color
use the style attribute `color` for adjusting the color of the icon, normally this will have the same color as the font in the paragraph
<Story name="Colors" args={{size:32, icon: 'person'}}>
  {(args) => {
    const colors = ['red', 'green', '#334455'];
    return (
      <Container>
        {colors.map(color => <Template key={color} {...args} color={color} />)}
      </Container>
    );
  }}
</Story>

### Size
use the style attribute `font-size` for adjusting the size of the icon. Normally the size is the same as the font size of the paragraph
<Story name="Size" args={{size:32, icon: 'person'}}>
  {(args) => {
    const sizes = [50, 40, 30, 20, 10];
    return (
      <Container>
        {sizes.map(size => <Template key={size} {...args} size={size} />)}
      </Container>
    );
  }}
</Story>
