### Ref

This component is created using `forwardRef` which exposes the `HTMLButtonCustomElement`.

```tsx
import Button, { HTMLButtonCustomElement } from '@equinor/fusion-react-button';
export const Component = () => {
  const btnRef = useRef<HTMLButtonCustomElement>(null);
  return <Button ref={btnRef} />
}
export default Component;
```
